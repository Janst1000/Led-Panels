InRange(A, n, k, low, high)
// we create a count array like in countsort
for i := 0 to k do
	C[i] := 0
for i := 1 to n do
	C[A[i]] = C[A[i]] + 1
for i := 2 to k do
	C [i] := C[i] + C[i - 1]
//then we just look subtract the lower counts from the upper counts and return
result := C[high] - C[low]
return result
